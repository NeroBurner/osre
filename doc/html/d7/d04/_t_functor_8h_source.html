125 template < class RET, class P1, class P2 >
126 class Functor {
127 public:
129  Functor()
130  : m_data( nullptr )
131  , m_refCounter( nullptr ) {
132  // empty
133  }
134 
136  Functor( const Functor& f )
137  : m_data( f.m_data )
138  , m_refCounter( f.m_refCounter ) {
139  ++(*m_refCounter);
140  }
141 
143  virtual ~Functor() {
144  if ( nullptr != m_refCounter ) {
145  if (--(*m_refCounter) <= 0 ) {
146  delete m_refCounter;
147  delete m_data;
148  }
149  }
150  }
151 
156  RET operator () (P1 p1, P2 p2) const {
157  // Check for a valid this pointer
158  if ( nullptr != m_data) {
159  return ( m_data->call( p1, p2 ) );
160  }
161 
162  // Just return the default
163  return RET();
164  }
165 
167  bool operator == (const Functor& other) const {
168  return (m_data == other.m_data);
169  }
170 
172  static Functor make(RET (*func) (P1,P2)) {
173  return Functor(new FunctorFunction<RET, P1, P2>(func));
174  }

178  static Functor Make(T* obj, RET (T::*func) (P1,P2)) {
179  return ( Functor( new FunctorMember<T, RET, P1, P2>( obj, func ) ) );
