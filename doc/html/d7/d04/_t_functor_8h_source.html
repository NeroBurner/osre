129 template < class RET, class P1, class P2 >
130 class Functor {
131 public:
133  Functor()
134  : m_data( nullptr )
135  , m_refCounter( nullptr ) {
136  // empty
137  }
138 
140  Functor( const Functor& f )
141  : m_data( f.m_data )
142  , m_refCounter( f.m_refCounter ) {
143  ++(*m_refCounter);
144  }
145 
147  virtual ~Functor() {
148  if ( nullptr != m_refCounter ) {
149  if (--(*m_refCounter) <= 0 ) {
150  delete m_refCounter;
151  delete m_data;
152  }
153  }
154  }
155 
160  RET operator () (P1 p1, P2 p2) const {
161  // Check for a valid this pointer
162  if ( nullptr != m_data) {
163  return ( m_data->call( p1, p2 ) );
164  }
165 
166  // Just return the default
167  return RET();
168  }
169 
171  bool operator == (const Functor& other) const {
172  return (m_data == other.m_data);
173  }
174 
176  static Functor make(RET (*func) (P1,P2)) {
177  return Functor(new FunctorFunction<RET, P1, P2>(func));
178  }

182  static Functor Make(T* obj, RET (T::*func) (P1,P2)) {
183  return ( Functor( new FunctorMember<T, RET, P1, P2>( obj, func ) ) );
184  }
185

192  // empty
193  }
