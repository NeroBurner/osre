100 void AbstractPlatformEventHandler::processEvents( Common::EventTriggerer *pTriggerer ) {
101  EventDataList *pList = getActiveEventDataList();
102  while( !pList->isEmpty() ) {
103  Common::EventData *pEventData = pList->front();
104  pTriggerer->triggerEvent( pEventData->getEvent(), pEventData );
105  pList->removeFront();
106  pEventData->release();
107  }
108  switchEventDataList();
109 }

118 EventDataList *AbstractPlatformEventHandler::getPendingEventDataList() {
